%% Locate something at a specific position:
%% #1 the thing to display
%% #2 the x-coordinate (between 0 and 1)
%% #3 the y-coordinate (between 0 and 1)
\gdef\locate#1#2#3{%
\setlength{\TPHorizModule}{\paperwidth}%
\setlength{\TPVertModule}{\paperheight}%%
\begin{textblock}{1}(#2,#3)#1\end{textblock}%
}%
%
\protected\gdef\tightboxB#1{\bgroup%
\fboxsep=0pt%
\fboxrule=2pt%
\mbox{\fcolorbox{black}{white}{\mbox{#1}}}%
\egroup}%
%
\protected\gdef\tightboxW#1{\bgroup%
\fboxsep=0pt%
\fboxrule=1pt%
\mbox{\fcolorbox{white}{white}{\mbox{#1}}}%
\egroup}%
%
\protected\gdef\igf#1#2#3#4{%
\locate{\tightboxW{\tightboxB{\includegraphics[page=#2,width=0.2\paperwidth]{#1.pdf}}}}{#3}{#4}%
}%
